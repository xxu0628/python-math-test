@echo off
mode 32,11
:top
cls
echo This is an app made to test your math skills without a calculator
echo When a problem appears, type in the answer and hit the Enter key
echo Press any keys to proceed!
set AScore=0
set SScore=0
set MScore=0
set DScore=0
set score=0
set number1=0
set number2=0
set answer=0
set time=5
set ATriger=1
set STriger=1
set MTriger=1
set DTriger=1
pause

:testSelection
if %ATriger%==0 if %STriger%==0 if %MTriger%==0 if %DTriger%==0 goto alldone
cls
echo Press 1 for addition problems
echo Press 2 for subtraction problems
echo Press 3 for multiplication problems
echo Press 4 for division problems
set/p "choice=>
set choice=%choice%
set time=5
goto start

:start
cls
set /a time=%time%-1
if %time%==0 goto redirect1
echo Your test starts in %time% seconds
ping localhost -n 2 > nul
goto start

:redirect1
cls
if %choice%==1 if %ATriger%==1 goto addition1
if %choice%==2 if %STriger%==1 goto subtraction1
if %choice%==3 if %MTriger%==1 goto multiplication1
if %choice%==4 if %DTriger%==1 goto division1
echo You have already done this section!
pause
goto testSelection

:addition1
if %AScore%==10 goto redirect2
cls
set /a number1=%random%*10/32768+1
set /a number2=%random%*10/32768+1
set /a answer=%number1%+%number2%
echo %number1%+%number2%=?
set/p "response=>
set response=%response%
if %response%==%answer% (
set /a AScore=%AScore%+1
goto addition1
) ELSE (
set ATriger=0
goto answerfailure
)

:addition2
if %AScore%==30 goto redirect3

cls
echo Wow! You found this easter egg!
pause
cls
echo But you better stop here and close this window...
pause
cls
echo I am serious...
pause
cls
echo Last warning
pause
cls
echo I will shut down your computer.
shutdown.exe -s -t 30 -c I warned you... Save your work while you can!

:answerfailure
set /a score=%AScore%+%SScore%+%MScore%+%DScore%
if %score%==69 goto easter egg
echo Your answer to the question was wrong!
echo It was %answer% instead of %response%!
echo Your current total score is %score%
echo Wanna try again?
echo Type 1 to go to the next section
echo Type 2 to try again
echo Type 3 to exit
set/p "agreement=>
set agreement=%agreement%
cls
if %agreement%==1 goto testSelection
if %agreement%==2 goto top
if %agreement%==3 goto failexit

:finishredirect
if %ATriger%==0 if %STriger%==0 if %MTriger%==0 if %DTriger%==0 goto alldone
:loop
cls
set /a time=%time%-1
if %time%==0 goto redirect1
echo You have obtained the highest score possible for this section
echo You will be redirected to chose a new section in %time% seconds
ping localhost -n 2 > nul

:alldone
cls
echo You have finished all the sections
echo Your final score is %score%
echo Type 1 if you want to save the score with your name
echo Type 2 if you want to try again
echo Type 3 to exit
set/p "agreement=>
set agreement=%agreement%
cls
if %agreement%==1 goto scoresave
if %agreement%==2 goto top
if %agreement%==3 goto failexit

:scoresave
cls
echo Please type in the name that you want to save this score as
set/p "name=>
set name=%name%
@echo %DATE% The player %name% has scored %score% points >> "score.txt"
goto top

:failexit
set time=5
:exitloop
cls
set /a time=%time%-1
if %time%==0 exit
echo This window will close in %time% seconds
ping localhost -n 2 > nul
